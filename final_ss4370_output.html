def projected_component(x, r=1):
    return np.dot(x, min((r/np.linalg.norm(x)), 1))

def derivative(x, ind, n):
    s = 0
    for i in range(n):
        s += np.subtract(x[ind], x[i])
        
    return 2*s

#starts with random vectors and approaches optimal configuration using projected gradient descent.
def find_n_points(n, d):
    X = np.random.rand(n,d)
    alpha = 0.001
    for i in range(50000):
        temp = []
        for j in range(n):
            temp.append(projected_component(gd(X[j], alpha, derivative(X,j, n))))  
        X = temp
    return X

def distance_between_points(x):
    temp = []
    for i in range(len(x)):
        for j in range(i+1,len(x)):
            temp.append(np.linalg.norm(np.subtract(x[i],x[j])))
    return temp

#computes angles between all the vectors in a given list of vectors. It uses the dotproduct to find the angle.
def compute_angle(vectors):
    angle = []
    for vector in vectors:
        temp = []
        for i in range(len(vector)):
            for j in range(len(vector)):
                if i!=j:
                    u,v = vector[j],vector[i]
                    temp.append(round(np.arccos(np.dot(u,v)/((np.linalg.norm(u))*(np.linalg.norm(v))))/math.pi*180,3))
        angle.append(temp)
    return angle
